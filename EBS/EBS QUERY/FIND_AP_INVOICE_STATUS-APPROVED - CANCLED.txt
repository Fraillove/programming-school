SELECT COUNT(*)
  
    INTO FL_COUNT_VALUE
  
    FROM AP_INVOICES_ALL S
  
   WHERE S.ATTRIBUTE_CATEGORY = L_ATTRIBUTE_CATEGORY
        
     and ATTRIBUTE7 = FL_BTA_NO
        
     AND ORG_ID = FL_ORG_ID
     AND INVOICE_NUM != INVOICE_NUMBER
    and APPS.Ap_Invoices_Pkg.GET_APPROVAL_STATUS(s.INVOICE_ID, 
s.INVOICE_AMOUNT,s.PAYMENT_STATUS_FLAG,s.INVOICE_TYPE_LOOKUP_CODE)!='CANCELLED';
